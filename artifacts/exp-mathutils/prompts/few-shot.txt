import static org.junit.jupiter.api.Assertions.assertDoesNotThrow;
import static org.junit.jupiter.api.Assertions.assertThrows;

import org.apache.commons.math3.util.MathUtils;
import org.junit.jupiter.api.Test;

public class MathUtilsFewShotExampleTest {

    @Test
    void testCheckFiniteWithFiniteValues() {
        double[] values = {1.0, 2.0, 3.0};
        assertDoesNotThrow(() -> MathUtils.checkFinite(values));
    }

    @Test
    void testCheckFiniteWithNaNValueThrowsException() {
        double[] values = {1.0, Double.NaN};
        assertThrows(Exception.class,
            () -> MathUtils.checkFinite(values));
    }

    @Test
    void testCheckFiniteWithInfiniteValueThrowsException() {
        double[] values = {Double.POSITIVE_INFINITY};
        assertThrows(Exception.class,
            () -> MathUtils.checkFinite(values));
    }
}

import static org.junit.jupiter.api.Assertions.assertArrayEquals;

import org.apache.commons.codec.binary.Hex;
import org.junit.jupiter.api.Test;

public class HexFewShotExampleTest {

    @Test
    void testEncodeHexWithSingleByte() {
        byte[] input = { 0x0A };
        char[] expected = { '0', 'a' };
        assertArrayEquals(expected, Hex.encodeHex(input));
    }

    @Test
    void testEncodeHexWithMultipleBytes() {
        byte[] input = { 0x00, 0x01 };
        char[] expected = { '0', '0', '0', '1' };
        assertArrayEquals(expected, Hex.encodeHex(input));
    }

    @Test
    void testEncodeHexWithZeroByte() {
        byte[] input = { 0x00 };
        char[] expected = { '0', '0' };
        assertArrayEquals(expected, Hex.encodeHex(input));
    }
}

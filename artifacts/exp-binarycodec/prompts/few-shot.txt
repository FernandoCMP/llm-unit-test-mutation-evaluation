import static org.junit.jupiter.api.Assertions.assertNotNull;
import static org.junit.jupiter.api.Assertions.assertEquals;

import org.apache.commons.codec.binary.BinaryCodec;
import org.junit.jupiter.api.Test;

public class BinaryCodecFewShotExampleTest {

    @Test
    void testToAsciiBytesWithSingleByteExample() {
        byte[] input = { 0x01 };
        byte[] result = BinaryCodec.toAsciiBytes(input);
        assertNotNull(result);
    }

    @Test
    void testToAsciiBytesWithEmptyArrayExample() {
        byte[] input = {};
        byte[] result = BinaryCodec.toAsciiBytes(input);
        assertEquals(0, result.length);
    }

    @Test
    void testToAsciiBytesWithMultipleBytesExample() {
        byte[] input = { 0x0F, (byte) 0xA5 };
        byte[] result = BinaryCodec.toAsciiBytes(input);
        assertNotNull(result);
    }
}

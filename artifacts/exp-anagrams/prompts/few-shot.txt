import com.thealgorithms.strings.Anagrams;
import org.junit.jupiter.api.Test;

public class AnagramsFewShotExampleTest {

    @Test
    void testApproach2WithAnagramStrings() {

        assertTrue(anagrams.approach2("listen", "silent"));
    }

    @Test
    void testApproach2WithDifferentLengths() {
        Anagrams anagrams = new Anagrams();
        assertFalse(anagrams.approach2("abc", "ab"));
    }

    @Test
    void testApproach2WithSameLengthNonAnagram() {
        Anagrams anagrams = new Anagrams();
        assertFalse(anagrams.approach2("hello", "world"));
    }
}

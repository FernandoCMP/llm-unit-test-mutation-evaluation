import static org.junit.jupiter.api.Assertions.assertNotNull;
import static org.junit.jupiter.api.Assertions.assertThrows;

import org.apache.commons.lang3.CharRange;
import org.junit.jupiter.api.Test;

public class CharRangeFewShotExampleTest {

    @Test
    void testContainsWithNormalRanges() {
        CharRange outer = CharRange.isIn('a', 'z');
        CharRange inner = CharRange.isIn('b', 'y');
        Boolean result = outer.contains(inner);
        assertNotNull(result);
    }

    @Test
    void testContainsWithNegatedOuterRange() {
        CharRange outer = CharRange.isNotIn('m', 'p');
        CharRange inner = CharRange.isIn('a', 'c');
        Boolean result = outer.contains(inner);
        assertNotNull(result);
    }

    @Test
    void testContainsWithNullRangeThrowsException() {
        CharRange outer = CharRange.isIn('a', 'z');
        assertThrows(IllegalArgumentException.class,
            () -> outer.contains(null));
    }
}

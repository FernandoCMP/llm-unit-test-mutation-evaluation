import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;

import org.junit.jupiter.api.Test;
import com.thealgorithms.maths.Area;

public class AreaFewShotExampleTest {

    @Test
    void testSurfaceAreaConeWithValidInput() {
        double result = Area.surfaceAreaCone(2.0, 3.0);
        assertEquals(Math.PI * 2.0 * (2.0 + Math.sqrt(13.0)), result);
    }

    @Test
    void testSurfaceAreaConeWithAnotherValidInput() {
        double result = Area.surfaceAreaCone(1.5, 4.0);
        assertEquals(Math.PI * 1.5 * (1.5 + Math.sqrt(18.25)), result);
    }

    @Test
    void testSurfaceAreaConeWithInvalidRadiusThrowsException() {
        assertThrows(IllegalArgumentException.class,
            () -> Area.surfaceAreaCone(0.0, 2.0));
    }
}

// Exemplos

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;

import org.apache.commons.lang3.Range;
import org.junit.jupiter.api.Test;

public class RangeFewShotExampleTest {

    @Test
    void testElementCompareToInsideRange() {
        Range<Integer> range = Range.between(1, 10);
        assertEquals(0, range.elementCompareTo(5));
    }

    @Test
    void testElementCompareToBeforeRange() {
        Range<Integer> range = Range.between(3, 8);
        assertEquals(-1, range.elementCompareTo(1));
    }

    @Test
    void testElementCompareToNullThrowsException() {
        Range<Integer> range = Range.between(2, 6);
        assertThrows(NullPointerException.class,
            () -> range.elementCompareTo(null));
    }
}
